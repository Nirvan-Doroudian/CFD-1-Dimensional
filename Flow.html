#include <iostream>
#include <fstream>

// Constants
const int numSpatialPoints = 50; // number of i's
const int numTemporalPoints = 150; // number of n's
const double dx = 2.0 / numSpatialPoints; //xmax = 2
const double dt = 0.5 / numTemporalPoints; //tmax = 0.5
const double speedBoundary = 1.0;
const double velocityInitial = 1.0;
const int xStart = 0;
const int xEnd = numSpatialPoints;
const int tStart = 0;
const int tEnd = numTemporalPoints;

// Function to write data to a text file
void writeDataToFile(std::ofstream& file, const double* data, int numPoints) {
    for (int i = 0; i < numPoints; ++i) {
        double x = i * dx;
        file << x << "\t" << data[i] << "\n";
    }
    file << "============================================" << "\n" << "============================================" << "\n";
}

int main() {
    std::ofstream file("flow_results.txt");
    file << "x or i \t" << "velocity \n";

    // Initialize arrays
    double velocities[numSpatialPoints];
    double newVelocities[numSpatialPoints];

    // Set initial conditions
    for (int i = 0; i < numSpatialPoints; ++i) {
        if (i >= 12 && i <= 25) {
            velocities[i] = 2.0;
        } else {
            velocities[i] = velocityInitial;
        }
    }

    // Solving the flow equation
    for (int t = tStart; t < tEnd; ++t) {
        // Apply boundary conditions
        velocities[xStart] = speedBoundary;
        velocities[xEnd - 1] = speedBoundary;

        // Update velocity at interior points
        for (int i = 1; i < numSpatialPoints - 1; ++i) {
            newVelocities[i] = velocities[i] - 0.5 * (dt / dx) * (velocities[i] - velocities[i - 1]);
        }

        // Update velocities array for the next time step
        for (int i = 1; i < numSpatialPoints - 1; ++i) {
            velocities[i] = newVelocities[i];
        }

        // Write data to the file
        writeDataToFile(file, velocities, numSpatialPoints);
    }

    file.close();
    std::cout << "Simulation completed. Results written to flow_results.txt" << std::endl;

    return 0;
}
